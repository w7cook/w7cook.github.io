main'1 = do

  test "substitute (\"x\", 5)" (substitute1 ("x", 5)) (Number 2)
  test "substitute (\"x\", 5)" (substitute1 ("x", 5)) x
  test "substitute (\"x\", 5)" (substitute1 ("x", 5)) (Add (Multiply x x) x)
  test "substitute (\"x\", 5)" (substitute1 ("x", 5)) (Add x y)
